'''
循环：
场景：
1、用户名和密码，反复输入
2、计算1-100
3、游戏死亡重生
4、...

方式：
1、while
2、for

while格式：
while 条件：
    代码

# 初始值
n = 1
while n <= 10:
    print(n)
    n += 1

# 打印1-50之间被3整除的数字
n = 1
while n <= 50:
    if n % 3 == 0:
        print(n)
    n += 1

# 打印1-10之间数字的累加和

num = 0

'''
for i in range(n)  -> 肯定是有固定次数的
    pass
while 条件:   ->  1、固定次数的循环   2、不确定的循环
    pass

while True: ->不确定次数的循环
    pass

掷骰子
两个：1-6
1、玩游戏要有金币，无金币无游戏
2、玩游戏赠金币1枚，充值获取金币
3、10元的倍数，充值，20个金币
4、玩游戏消耗5个金币
5、制定输赢，猜大小（两个点数相加，超过6则是大，否则为小），猜对有2枚鼓励金币，猜错无奖励
6、游戏结束：1、主动退出   2、没有金币退出
7、只要退出，则打印金币数，共玩了几局
'''
import random

# 金币数
coins = 0
# 计数器
count = 0
if coins < 5:
    print('金币不足，请充值再玩！')
    while True:
        money = int(input('请输入充值金额:'))
        # 10元的倍数，充值，20个金币
        if money % 10 == 0:
            coins += money // 10 * 20
            print('充值成功，当前金币有%d个' % coins)
            # 开启游戏之旅
            print('--------------开启游戏之旅--------------')
            ans = input('是否开始游戏(y/n)?')
            while coins > 5 and ans == 'y':
                # 先扣金币
                coins -= 5
                # 赠送金币
                coins += 1
                # 产生两枚随机骰子数
                ran1 = random.randint(1, 6)
                ran2 = random.randint(1, 6)
                # 猜大小
                guess = input('洗牌完毕，请猜大小:')
                # 判断大小
                if (ran1 + ran2 > 6 and guess == '大') or (ran1 + ran2 <= 6 and guess == '小'):
                    print('恭喜猜对了，获得2枚金币鼓励')
                    coins += 2
                else:
                    print('抱歉猜错了')
                count += 1
                ans = input('是否继续游戏(y/n)?')
            # 打印玩的次数和剩余金币数
            print('共玩了%d次游戏，还剩余%d个金币' % (count, coins))
            break
        else:
            print('不是10的倍数，充值失败！')

# 3.3跳出语句

'''
break
continue
都是出现在循环中的

不同：
break:跳出循环结构
continue:跳过本次循环（后面的语句不执行）继续下一次循环
'''
# 不打印能被3整除的
# for i in range(10):
#     if i % 3 != 0:
#         print(i)

for i in range(10):
    if i % 3 == 0:
        continue
    print(i)

# 3.4循环嵌套

'''
循环嵌套：
if 条件：
    if  条件：
        pass
else:
    if  条件：
        pass

while 循环:
*
**
***
****
*****
'''
# 单循环
n = 1
while n <= 5:
    print('*' * n)
    n += 1

# 双循环
n = 1
while n <= 5:
    m = 0
    while m < n:
        print('*', end='')
        m += 1
    print()
    n += 1

'''
*****
****
***
**
*
'''

n = 5
while n >= 1:
    print('*' * n)
    n -= 1

n = 5
while n >= 1:
    m = n
    while m >= 1:
        print('*', end='')
        m -= 1
    print()
    n -= 1

for i in range(6):
    print('*' * i)
