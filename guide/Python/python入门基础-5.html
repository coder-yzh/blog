冒泡排序
'''

import random

dig = []
for i in range(8):
    ran = random.randint(1, 100)
    dig.append(ran)

for i in range(len(dig) - 1):  # 轮数
    for j in range(len(dig) - 1 - i):
        if dig[j] > dig[j + 1]:  # 两两比较
            dig[j], dig[j + 1] = dig[j + 1], dig[j]

print(dig)

# 5.4列表练习

'''
王者荣耀角色管理

角色：姓名，性别，职业

添加角色
删除角色
修改角色
查询角色 单个角色
显示所有角色
退出管理系统
'''
import time
import random

all_role = []  # 存放所有角色的'大容器'
print('~~~~~~~~~欢迎进入王者荣耀角色管理~~~~~~~~~')

while True:
    choice = input('请选择功能:\n1.添加角色\n2.删除角色\n3.修改角色\n4.查询角色\n5.显示所有角色\n6.退出管理系统\n')
    # 判断
    if choice == '1':
        print('添加角色模块：')
        name = input('角色名：')
        sex = input('性别：')
        job = input('职业：')
        role = [name, sex, job]
        # 添加到all_role
        all_role.append(role)
        print('\t成功添加{}到王者荣耀系统！\n'.format(name))
    elif choice == '2':
        print('删除角色模块：')
        role_name = input('输入角色名：')
        # 查找是否存在此角色名称
        for role in all_role:
            if role_name in role:
                # 添加一个是否确认删除询问，确认删除则再删除
                s = input('是否确认删除角色：{}(y/n)'.format(role_name))
                if s == 'y':
                    all_role.remove(role)
                    print('成功删除角色：{}'.format(role_name))
                    break
                else:
                    print('取消删除角色：{}'.format(role_name))
        else:
            print('本系统不存在角色：{},请检查角色名称.'.format(role_name))
    elif choice == '3':
        print('修改角色模块：')
        role_name = input('输入角色名：')
        # 查找是否存在此角色名称
        for role in all_role:
            if role_name in role:
                # 添加一个是否确认修改询问，确认修改则再修改
                s = input('是否确认修改角色：{}(y/n)'.format(role_name))
                if s == 'y':
                    role[0] = input('请输入修改后的角色名：')
                    role[1] = input('请输入修改后的性别：')
                    role[2] = input('请输入修改后的职业：')

                    print('成功修改角色：{}'.format(role_name))
                    break
                else:
                    print('取消修改角色：{}'.format(role_name))
        else:
            print('本系统不存在角色：{},请检查角色名称.'.format(role_name))
    elif choice == '4':
        print('查询角色模块：')
        role_name = input('输入角色名：')
        # 查找是否存在此角色名称
        for role in all_role:
            if role_name in role:
                print('\t存在此角色信息如下:')
                print('\t{}{}{}'.format(role[0].center(10), role[1].center(10), role[2].center(10)))
                break
        else:
            print('本系统不存在角色：{},请检查角色名称.'.format(role_name))
    elif choice == '5':
        print('显示所有角色模块~~~~')
        print('{}{}{}'.format('名称'.center(10), '性别'.center(10), '职业'.center(10)))
        for role in all_role:
            print(role[0].center(10), end='')
            print(role[1].center(10), end='')
            print(role[2].center(10))
    elif choice == '6':
        print('正在退出王者荣耀管理系统~~~~')
        time.sleep(3)
        print('成功退出！')
        break
    else:
        print('输入错误，请重新选择！')
