# 1.3数据类型

'''
数据类型：
int
float
string
boolean
'''

# 通过type(变量名)，输出变量类型

money = 28  # 声明了一个名称叫money的变量，赋值为28
# <class 'int'>     表示整型
print(type(money))  # print()属于一个内置函数，负责输出结果

# money是一个变量，后面的值允许发生变化

money = 28.9
# <class 'float'>   表示浮点类型
print(type(money))

money = '1万'
# <class 'str'> 表示字符串
print(type(money))
money = "1万"
# <class 'str'> 表示字符串
print(type(money))
money = '''1万'''
# <class 'str'> 表示字符串
print(type(money))

# 单双引号互相嵌套使用
message = 'y说："今天下暴雨了！"'
print(message)
message = "y说：'今天下暴雨了！'"
print(message)

# 保留格式输出
shige = '''
            静夜思
            唐 李白
      床前明月光，疑是地上霜。
      举头望明月，低头思故乡。
      
'''
print(shige)

# 布尔类型：True False
# 开发中判断，比如：是否登录成功...
isLogin = True  # 真
print(isLogin)
isLogin = False  # 假
print(isLogin)

print(type(isLogin))  # <class 'bool'>  布尔类型

#类型转换练习

'''
键盘输入两个数，输出差和
'''

one = input('输入第一个数：')  # 100
two = input('输入第二个数：')  # 50

# 进行计算
print(one + two)
# 转换
print(float(one) + float(two))  # 150.0

print(int(one) + int(two))  # 150

# 差
print(int(one) - int(two))  # 50

print(float(one) - float(two))  # 50.0

'''
以变量名：a
str ->  int     int(a)  但是如果'9.9'而且是字符串类型转成int的时候报错了
str ->  float   float(a)

int ->  str     str(a)
float   ->  str str(a)

int ->  float   float(a)
float   ->  int int(a)  只不过float类型中小数点后面的数字被抹掉了
'''

flag = True
# bool -> int       True->1 False->0
print(int(flag))
print(float(flag))
print(str(flag))  # 'True'

'''
思考：
a=5
能否将a转成bool类型?
'''
a = 5
print(bool(a))
a = 0
print(bool(a))
a = ''
print(bool(a))

# 当变量的值是0或者空字符串时，转换结果是False,其他只要有值则为True

# 1.5输出格式化

name = '蔡徐坤'
age = 26
# 我喜欢听26岁的蔡徐坤唱歌!
print('我喜欢听' + str(age) + '岁的' + name + '唱歌!')
# 字符串如何进行格式化？
'''
符号：
%s  字符串 string
%d  整数  digit
%f  浮点数 float
%c  字符  char
%%  输出%号
'''
print('我喜欢听%d岁的%s唱歌!' % (age, name))

money = 999.95

# 26岁的蔡徐坤一首歌挣了999.95块钱
print('%d岁的%s一首歌挣了%f块钱' % (age, name, money))

print('%s岁的%s一首歌挣了%s块钱' % (age, name, money))  # 进行了强制类型转换

print('%d岁的%s一首歌挣了%.2f块钱' % (age, name, money))  # 小数保留两位小数

# 1.6运算符

'''
运算符：
=赋值运算符
算术运算符：  + - * / % // **

'''

a = 1
b = 2

c = a + b
print(a, b, c, 100, 100, sep='#')
# 1#2#3#100#100
print(a, b, c, end='\n')  # 1 2 3 \n,不加end也可以，因为默认有换行符

print(c - a)
print(c * a)
print(c / 2)  # 普通的除法
print(c // 2)  # 整除，结果为一个整数
print(2 ** 3)  # m**n    表示m的n次方

print(13 % 5)  # 取模，取余数，可用来判断奇偶数

'''
键盘输入一个3位数的整数，打印个位 十位 百位
'''
num = int(input('请输入一个3位整数'))
print('个位数：', num % 10)
print('十位数：', num // 10 % 10)
print('百位数：', num // 100)
 
    
'''
赋值运算符：
=
+=
-=
*=
/=
//=
**=
%=
'''
a = 8
b = 4
c = 0
# c = a + 1

a += 1  # 等同于a=a+1
print(a, b, c)

a -= b
print(a, b, c)

b += a
print(a, b, c)

d = 3
b //= d
print(a, b, c, d)



'''
关系运算符：  结果：True False
> < >= <= == != is

'''

a = 10
b = 23
print(a > b)  # False
print(a < b)  # True

print(a == b)  # False
print(a != b)  # True

x = 'abc'
y = 'abcd'
print(x == y)  # False
print(x < y)  # True

c = 10
print(a >= c)  # True

'''
输入考试分数，判断成绩是否在100到80之间?
'''
score = float(input('请输入成绩:'))
print(80 <= score <= 100)  # python可以直接按照分数段来进行书写
'''
以下为错误写法：
if (score >= 80 && score <= 100)

'''
输入华为手机价格，输入苹果手机价格，判断是否相等
'''
hw_price = float(input('输入华为手机价格:'))
pg_price = float(input('输入苹果手机价格:'))
print(hw_price == pg_price)


'''
逻辑运算符:
and or not

and:与   并且
A and B

or:或    或者
A or B

not：
not True->False
not False->True

'''
a = 1
b = 3
print(b and a)  # and 两边都是非0数字，结果是最后的数字值

c = 0
print(c and a)  # and两边只要有一边为0，结果即为0

print(a > c and a < b)  # True
print(a == c and a < b)  # False

# and应用场景：
# 1、username and password

print('-' * 20)  # 打印20个-号

print(b or a)
print(a or b)  # or 两边都是非0数字，结果是前面的数字值

print(c or b)

print(a > c or a < b)  # True
print(a == c or a < b)  # True
# or应用场景：
# 登陆时使用： 1、账号密码 or 2、手机号码 验证码

print('-' * 20)  # 打印20个-号

flag = True
print(not flag)
print(not (a > c))  # False
