# 9.4文件操作-复制

# 文件的复制
'''
原文件： D:\work\logo.jpg
目标文件： D:\work1\logo.jpg

with 结合open使用，可以帮助我们自动释放资源

'''
# stream = open(r'D:\work\logo.jpg', 'rb')

with open(r'D:\work\logo.jpg', 'rb') as stream:
    container = stream.read()  # 读取文件内容

    with open(r'D:\work1\logo.jpg', 'wb') as wstream:
        wstream.write(container)

print('文件复制完成！')


#---------------分割线------------------
import os

# 文件复制
src_path = r'D:\work'
target_path = r'D:\work1'


# 封装成函数
def copy(src, target):
    if os.path.isdir(src) and os.path.isdir(target):
        filelist = os.listdir(src)  # ['aa.txt','','']

        for file in filelist:  # aa.txt

            path = os.path.join(src, file)
            with open(path, 'rb') as rstream:
                container = rstream.read()

                path1 = os.path.join(target,file)
                with open(path1,'wb') as wstream:
                    wstream.write(container)
        else:
            print('复制完毕！')


# 调用函数
copy(src_path,target_path)

#---------------分割线------------------
#对文件中的文件，文件夹一块复制
import os

src_path = r'D:\work'
target_path = r'D:\work3'


def copy(src_path, target_path):  # 'D:\work'
    # 获取文件夹里面内容
    filelist = os.listdir(src_path)  # ['a1.doc', 'aa.txt', 'logo.jpg', 'WeChat']
    # print(filelist)
    # 变量列表
    for file in filelist:
        # 拼接路径
        path = os.path.join(src_path, file)  # D:\work\..
        # 判断是文件夹还是文件
        if os.path.isdir(path):  # path = ‘D:\work\..’
            # 递归调用copy
            target_path1 = os.path.join(target_path, file)
            # 创建文件夹
            os.mkdir(target_path1)
            copy(path, target_path1)

        else:
            # 不是文件夹则直接进行复制
            with open(path, 'rb') as rstream:
                container = rstream.read()

                path1 = os.path.join(target_path, file)
                with open(path1, 'wb') as wstream:
                    wstream.write(container)
    else:
        print('复制完成！')


# 调用copy
copy(src_path, target_path)
