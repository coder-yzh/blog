# 8.2闭包

'''
嵌套：

闭包：

搜索规则：
内层函数->外层函数->全局->系统builtins
'''

#
# def outer():
#     print('outer---hello')
#     a = 100
#
#     def inner():  # inner这个函数是在outer函数内部定义的
#         b = 200
#         # b += a  # 内部函数可以使用外部函数的变量
#         # a += b  # 会报错，无权限修改，需要 使用nonlocal a
#         nonlocal a  # 如果想修改外部函数的变量，需要在内部函数中添加：nonlocal
#         a += b
#         print('我是内部函数,inner---hello', b, a)
#
#     # result = locals()  # 表示查看函数中的局部变量，以字典的形式返回
#     # print(result)  # {'a': 100, 'inner': <function outer.<locals>.inner at 0x000002877CCAB5E0>}
#     print(a)
#     # print(inner)  # <function outer.<locals>.inner at 0x000001762809B5E0>
#     inner()  # inner函数只在outer函数内部可见
#
#
# outer()
# # inner()    #这里会报错，在outer函数外部无法访问到inner函数

a = 100


def outer():
    a = 200

    def inner():
        # a = 300
        nonlocal a
        a -= 50
        print('内部函数：', a)

    print(a)
    inner()


# 200 150 100

outer()
print(a)

'''
闭包：
1、嵌套函数
2、内部函数引用了外部函数的变量
3、返回值是内部函数

闭包=函数块+引用环境

闭包应用  装饰器用
'''


def outer(n):
    a = 10

    def inner():
        b = a + n
        print('内部函数', b)

    return inner


r = outer(5)
print(r)  # <function outer.<locals>.inner at 0x0000021FC0EEB5E0>

r()  # 内部函数 15

# 8.3 装饰器-定义

'''
装饰器：
遵循开放封闭原则，在不改变原函数的情况下，扩展了函数得功能

定义：
def XXX(func):
    def XXX(参数,...):
        ...
        func()
        ...
        return yyy
    return XXX

装饰：
@装饰器名   ->  原函数=装饰(原函数)
def 原函数():
    pass

装饰器功能：
引入日志
函数执行时间统计
执行函数前预备处理
执行函数后清理功能
权限校验等场景
缓存
'''


# 定义装饰器
def decorater(func):
    print('------>1')

    def wrapper():
        func()
        print('刷墙漆')
        print('铺地板')
        print('买家具')
        print('精装修房子')

    print('------>2')
    return wrapper


@decorater  # house=decorator(house)
def house():
    print('毛坯房...')


house()

# 8.4 装饰器-带参

'''
带参数的装饰器
如果原函数有参数则装饰器内部函数也要有参数
'''


def decorater(func):
    def wrapper(*args, **kwargs):  # address='北京四合院'
        # args是一个元组，多参数用*args
        func(*args, **kwargs)  # func就是house
        print('刷墙漆')
        print('铺地板')
        print('买家具')
        print('精装修房子')

    return wrapper


@decorater
def house(address):
    print('房子的地址在：{}是一个毛坯房...'.format(address))


@decorater
def changfang(address, area):
    print('房子的地址在：{}是一个毛坯房，建筑面积是：{}平米'.format(address, area))


@decorater
def hotel(name, address, area=40):
    print('酒店的名字是：{}，酒店的地址在：{}是一个毛坯房，建筑面积是：{}平米'.format(name, address, area))


house('北京四合院')  # house就是wrapper

print('-' * 50)

changfang('北京沙河北大桥', 1000)

print('-' * 50)

hotel(name='全季大酒店', address='北京国贸')

# 8.5装饰器-返回值

'''
装饰器修饰有返回值函数
原函数有返回值，装饰器的内部函数也要有返回值
'''


def decorater(func):  # function 函数
    def wrapper(*args, **kwargs):
        # args是一个元组，多参数用*args
        r = func(*args, **kwargs)  # func就是house
        print('预计装修费用是：{}元'.format(r))
        print('刷墙漆')
        print('铺地板')
        print('买家具')
        print('精装修房子')
        return 60000

    return wrapper


@decorater
def house():
    print('我是一个毛坯房...')
    return 50000


r = house()  # house就是wrapper
print(r)

# reduce(): 对列表中的元素进行加减乘除运算的函数
from functools import reduce

result = reduce(lambda x, y: x + y, tuple1)

# -----------------------分割线----------------------
tuplex = (5,)
result = reduce(lambda x, y: x + y, tuplex)

print(result)  # 5 ，这是因为reduce中还有一个initial参数，默认为0

result = reduce(lambda x, y: x + y, tuplex, 10)

print(result)  # 15 ，这是因为reduce中还有一个initial参数，设置为10

# -----------------------分割线----------------------

tuple2 = (1, 2, 3)

result = reduce(lambda x, y: x - y, tuple2, 10)
print(result)  # 4 ，这是因为reduce中还有一个initial参数，设置为10,10-1-2-3=4

# 8.8系统高阶函数

'''
高阶函数：
一个函数的参数是另一个函数，高阶函数
系统高阶函数：max  min sorted

'''
from functools import reduce

m = max(5, 9)
print(m)

m = max([2, 4, 67, 8, 9])
print(m)

list1 = [('tom', 19), ('tony', 20), ('lily', 18), ('daniel', 21), ('rose', 22)]
m = max(list1, key=lambda x: x[1])
print(m)

mi = min(list1, key=lambda x: x[1])
print(mi)

s = sorted(list1, key=lambda x: x[1])
print(s)  # [('lily', 18), ('tom', 19), ('tony', 20), ('daniel', 21), ('rose', 22)]

s = sorted(list1, key=lambda x: x[1], reverse=True)
print(s)  # [('rose', 22), ('daniel', 21), ('tony', 20), ('tom', 19), ('lily', 18)]

# filter的匿名函数要求返回值必须是bool类型，只有bool类型结果为True的才是符合过滤条件的
rr = filter(lambda x: x[1] > 20, list1)
print(rr)  # <filter object at 0x000001ED28806FA0>
print(list(rr))  # [('daniel', 21), ('rose', 22)]

ma = map(lambda x: x[1], list1)
print(ma)  # <map object at 0x000002D5E97F6F40>
print(list(ma))

# 提取名字，并且首字母大写            通过匿名函数指明提取的内容，并对内容进行加工
ma = map(lambda x: x[0].title(), list1)
print(list(ma))  # ['Tom', 'Tony', 'Lily', 'Daniel', 'Rose']

# reduce
r = reduce(lambda x, y: x * y, [1, 2, 3, 4, 5])
print(r)  # 5!

r = reduce(lambda x, y: x + y, [1, 2, 3, 4, 5])
print(r)  # 累加和
