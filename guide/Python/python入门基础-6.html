'''

结构：{键1:值,键2:值,键3:值}
字典：{}
元素：键值对 注意：键是唯一的，值是允许重复的

下标 或者 切片 -> 没有

1、添加元素：
字典名[key]=value
注意：key是唯一的，所以在添加的时候如果出现同名的key，后面key对应的value则替换原来的

2、修改value值
字典名[key] = value

关键看’键‘： 如果字典中不存在键则是添加
           如果字典中存在键，则是修改
'''

dict1 = {}
print(type(dict1))  # <class 'dict'>

# {'name': '阿泽'}
dict1['name'] = '阿泽'
print(dict1)

# {'name': '阿泽', 'age': 20, 'sex': '男'}
dict1['age'] = 20
dict1['sex'] = '男'
print(dict1)

# 改变年龄 键可以添加，删除，但是不能修改，只有键后面的值可以修改
dict1['age'] = 21
print(dict1)

# 做添加
dict1['score'] = 90

# 6.4集合

'''
集合：底层是字典
set
特点：没有重复，无序的（没有下标）
符号：{}      {元素，元素，元素，...}   ->  集合
     {}      {key:value,...}        ->  字典

定义集合：

'''

set1 = {'张三'}
print(type(set1))

#
list = [1, 3, 6, 8, 9, 9, 1, 2, 3, 9]
set2 = set(list)
print(set2)  # {1, 2, 3, 6, 8, 9}

# 声明空集合只能这么声明，才表示的是集合
set3 = set()  # <class 'set'>
print(type(set3))
print(len(set3))  # 0

set3.add('三体')

set3.add('盗墓笔记')

set3.add('红楼梦')

# set:无序，不允许重复

print(set3)

set3.add('三体')

        code += s[index]
    # 将code添加到set中
    code_list.add(code)
    # 判断长度
    if len(code_list) == 5:
        break

# 6.5练习

'''
图书管理系统
至少5那本书
library=[{'bookname':'xxx','author':'xxx','price':1000,'number':4},{},{}]
要求：
书名可以一样，作者名不同
1、借书
2、还书
3、查询（书名/作者）
4、查看所有
5、退出
'''
import time

library = [{'bookname': '西游记', 'author': '吴承恩', 'price': '100', 'number': '0'},
           {'bookname': '水浒传', 'author': '施耐庵', 'price': '100', 'number': '40'},
           {'bookname': '红楼梦', 'author': '曹雪芹', 'price': '100', 'number': '40'},
           {'bookname': '天珠变', 'author': '唐三少', 'price': '100', 'number': '40'},
           {'bookname': '西游记', 'author': '吴成恩', 'price': '100', 'number': '40'}]

book_bro = []
while True:
    choice = input('1、借书\n2、还书\n3、查询\n4、查看所有\n5、退出\n请输入您的选择：')
    if choice == '1':
        # 1、借书模块
        book_name = input('请输入书名：')
        # 查询书名
        for i in library:
            # 判断书名是否存在
            if i['bookname'] == book_name:
                if int(i['number']) >= 1:
                    print('借书成功')
                    # book_bro保存借的书信息
                    book_bro.append(i)
                    # 将借的书从书库删除
                    i['number'] = str(int(i['number']) - 1)
                    break
                else:
                    print('余书容量不足，借书失败！')
        else:
            print('查无此书，借书失败')
    elif choice == '2':
        # 2、还书模块
        book_name = input('请输入书名：')
        # 查询书名
        for i in book_bro:
            # 判断书名是否存在
            if i['bookname'] == book_name:
                print('还书成功')
                # 将还的借的书保存至图书馆
                for j in library:
                    if j['bookname'] == i['bookname'] and j['author'] == i['author']:
                        j['number'] = str(int(j['number']) + 1)
                        break
                # 将借的书从借书单中删除
                book_bro.remove(i)
                break
        else:
            print('查无此书，还书失败')
    elif choice == '3':
        # 3、查询模块
        choice_chaxun = input('查询书名(1)or查询作者(2)：')
        if choice_chaxun == '1':
            flag = 0
            book_name = input('请输入书名：')
            for i in library:
                # 判断书名是否存在
                if i['bookname'] == book_name:
                    flag += 1
                    if flag == 1:
                        print('{}{}{}{}'.format('bookname'.center(10), 'author'.center(10), 'price'.center(10),
                                                'number'.center(10)))
                    print('{}{}{}{}'.format(i['bookname'].center(10), i['author'].center(10), i['price'].center(10),
                                            i['number'].center(10)))
            else:
                print('查无此书')
        elif choice_chaxun == '2':
            flag = 0
            author_name = input('请输入作者：')
            for i in library:
                # 判断书名是否存在
                if i['author'] == author_name:
                    flag += 1
                    if flag == 1:
                        print('{}{}{}{}'.format('bookname'.center(10), 'author'.center(10), 'price'.center(10),
                                                'number'.center(10)))
                    print('{}{}{}{}'.format(i['bookname'].center(10), i['author'].center(10), i['price'].center(10),
                                            i['number'].center(10)))
                    break
            else:
                print('查无此书')
        else:
            print('输入有误')
    elif choice == '4':

# 6.6公共方法

list1 = [1, 2, 3, 5, 17, 90, 0]
result = max(list1)  # 最大值
print(result)

result = min(list1)  # 最小值
print(result)

result = sum(list1)  # 求和
print(result)

result = abs(-1)  # 绝对值
print(result)

t1 = (45, 12, 78, 90, 10, 55)

# 排序
result = sorted(t1)
print(result)  # 返回的是列表[10, 12, 45, 55, 78, 90]
result = sorted(t1, reverse=True)
print(result)  # 返回的是列表[90, 78, 55, 45, 12, 10]

# 给ASSCII码值返回该值对应的字符
result = chr(65)
print(result)

# 给字符返回该值对应的ASSCII码值
result = ord('A')
print(result)

# * 支持 字符串、列表、元组
# - & | 支持 集合
# in 或者not in 支持 字符串、列表、元组、字典（键key）
list2 = [1, 2]
list2 = list2 * 4
print(list2)
