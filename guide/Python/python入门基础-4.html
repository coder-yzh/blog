# 4.3字符串切片

'''
切片：字符串，列表
格式：字符串变量[start:end:step]，默认从左往右一个一个取元素
step:
1、步长
2、方向 step 正数 从左往右
       step 负数 从右往左
'''

s = 'ABCDEFG'
print(s[1:4])  # BCD
print(s[0:5])  # ABCDE
print(s[:5])  # ABCDE 默认从0开始，从0到index=4
print(s[-3:7])  # EFG,索引机制规则可以交叉使用
print(s[-3:])  # EFG 从-3开始到结尾

print(s[:])  # 打印出的结果与print(s)是一样的
print(s)

print(id(s[:]), id(s))  # 地址一样，内容一样

print(s[1:-1])  # 不要两头，要中间
print(s[2:-2])

print('*' * 20)

print(s[:-1:2])  # ACE 隔一个取一个，步长是2
print(s[1::2])  # BDF
print(s[::4])  # AE
print(s[-3::-4])  # EA

print('*' * 20)

print(s[::-1])  # 倒着输出
print(s)
print(s[::-2])  # step代表步长和方向
print(s[6:0:-2])  # GEC
print(s[6::-2])  # GECA

# 4.4字符串常见操作

'''
字符串常见操作
https://mbd.baidu.com/newspage/data/mdpage?tag=8&id=7714

find：从左往右查找，只要遇到一个符合要求的则返回位置，如果没有找到符合要求的则返回-1
rfind：从右向左查找，要遇到一个符合要求的则返回位置，如果没有找到符合要求的则返回-1
count：统计指定字符的个数
index与find的区别：index也是表示查找，但是如果找不到会报错
'''
path = 'https://mbd.baidu.com/newspage/data/mdpage?tag=8&id=7714'

# 查找 find,index,rfind,rindex
i = path.find('?')
data_name = path[i + 1:]
print(i)
print(data_name)

# 查找失败时返回-1
i = path.find('#')
print(i)

# rfind  right find 找后缀
i = path.rfind('/')
zhui = path[i:]
print(zhui)

'''
查找字符串有几个点？
'''
n = path.count('.')
print(n)

i = path.find('tag')
print(i)

'''
判断：startswith,endswith,isalpha,isdigit,isalnum,isspace
返回值都是布尔类型,(True,False)
startswith:判断是否以xxx开头的
endswith:判断是否以xxx结尾的
'''
s = 'mdpage?tag=8&id=7714'
result = s.startswith('md')  # 判断是否以md开头的
print(result)

result = s.endswith('=7714')    # 判断是否以=7714结尾的
print(result)

s = 'HELLO'
result = s.isalpha()  # 字母组成
print(result)

s = '100'
result = s.isdigit()  # 数字组成
print(result)

s = 'A1234'
result = s.isalnum()  # 字母数字组合
print(result)

s = '  '
result = s.isspace()  # 空格组合,True
print(result)

s = ''
result = s.isspace()  # 空格组合，空白是False
print(result)

s = 'HELLO'
result = s.isupper()  # 大写字母组合
print(result)

s = 'hello'
result = s.islower()  # 小写字母组合
print(result)

'''
替换内容：replace(old,new,count) 默认全部替换，也可以通过count指定次数

切割字符串：split,rsplit,splitlines,partition,rpartition

split('分隔符',maxsplit) 返回的结果是一个列表(list),maxsplit表示最多的分割次数
同理：rsplit，从右往左切割

splitlines()一行一行的截取，返回的结果是一个列表(list)

partition('分隔符')查找分隔符，按照分隔符分割字符串，返回的结果是一个列表(list)

修改大小写：capitalize,title,upper,lower
title()单词第一个字母大写
upper()所有字母大写
lower()所有字母小写
capitalize()第一个字母大写
'''

# 替换李泽，阿硕   用正则表达式 循环+列表 （待解决）
# replace()
s = '李泽说：阿硕你来唱歌吧！阿硕你来跳舞吧'
result = s.replace('阿硕', '**', 1)
print(result)

# 切割字符串：split,rsplit,splitlines,partition,rpartition
# split and rsplit
s = '李泽 阿硕 郭琦'
result = s.rsplit(' ', 1)
print(result)

# splitlines
s = '''阿硕你来唱歌吧！
阿硕你来跳舞吧！
阿硕你来学习吧！
阿硕你来吃饭吧！
'''
result = s.splitlines()
print(result)

# partition
s = '李泽 阿硕 郭琦'
result = s.partition(' ')
print(result)

# 修改大小写：capitalize,title,upper,lower
s = 'hello world'
result = s.title()
print(result)

result = s.upper()
print(result)

result = s.lower()
print(result)

result=s.capitalize()
print(result)

'''
空格处理：ljust,rjust,center,lstrip,rstrip,strip
字符串拼接：join

lstrip,rstrip,strip：去除（所有/左侧/右侧）空格

ljust,rjust,center：添加空格，控制字符串的对齐方式，左对齐，右对齐，居中

'''
# username = input('用户名:')
# print(len(username))

# strip 去除空格
s = '  admin    '
print(len(s))
result = s.strip()
print(len(result))
result = s.rstrip()
print(len(result))
result = s.lstrip()
print(len(result))

# center  居中
s = 'hello world'
result = s.center(30)
print(result)

# ljust 左对齐
result = s.ljust(30)
print(result)
# rjust 右对齐
result = s.rjust(30)
print(result)

# 4.5字符串练习

'''
模拟文件上传，键盘输入上传文件的名称(abc.jpg)
判断文件名是否大于6位以上，扩展名是否是:jpg,gif,png格式
如果不是则提示上传失败
如果名字不满足条件，而扩展名满足条件则随机生成一个6位数字组成的文件名，打印成功上传xxx.png
'''
import random

'''
产生随机字母和数字的组合
'''
filename = ''
s = 'QWERTYUIOPASDFHJKLZXCVBNMqwertyuiopasdfghjklzxcvbnm0123456789'
for i in range(6):
    index = random.randint(0, len(s) - 1)  # 0开始 随机产生下标
    filename += s[index]  # 获取下标对应的字符
print(filename)

s = input('请输入上传文件的名称：')
# 判断扩展名
if s.endswith('jpg') or s.endswith('gif') or s.endswith('png'):
    # 判断文件名字
    i = s.rfind('.')
    name = s[:i]
    if len(name) < 6:
        # 重新构建名字
        # ans = str(random.randint(100000, 999999))
        ans = filename
        ans = ans + s[i:]
        print('成功上传%s' % ans)
    else:
        print('成功上传%s' % s)
else:
    print('上传失败')

'''
admin123 15811119999 210729
用户名或者手机号码登录+密码
用户名：全部小写，首字母不能是数字，长度必须6位以上
手机号码：纯数字 长度11

密码：必须是6位数字

以上符合条件则进入下层验证：
判断用户名+密码是否正确
'''
'''
如下伪代码：
while True:
    name = input('用户名/手机号:')
    if 验证是否符合用户名的规则 or 验证是否符合手机号码的规则:
        # 继续输入密码
        password = input('密码:')
        # 判断密码是否是6位数字
        if 判断是否是6位数字:
            # 验证name+密码，正确性
            pass
        else:
            print('密码必须是6位数字')

    else:
        print('用户名或手机号码，密码输入格式错误')
'''
flag = True
while flag:
    name = input('用户名/手机号:')
    while (name.isalnum() and not name[0].isdigit() and len(name) >= 6 and name.islower()) or (
            name.isdigit() and len(name) == 11):
        # 继续输入密码
        password = input('密码:')
        # 判断密码是否是6位数字
        if password.isdigit() and len(password) == 6:
            # 验证name+密码，正确性
            if (name == '15811119999' or name == 'admin123') and password == '210729':
                print('登陆成功!')
                flag = False
                break
            else:
                print('有户名或者密码有误!')
                break
        else:
            print('密码必须是6位数字')

    else:
        print('用户名或手机号码输入格式错误')
